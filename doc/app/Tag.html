def self.createTags(body,post_id)
  tags = body.scan(/#[a-zA-Z0-9]+/).uniq
  tags.each do |name|
    name = name.downcase
    tag = where(name: name)
    tag_id = nil
    if tag == []
      t = Tag.create(name: name, count: 1)
      tag_id = t.id
    else
      count = tag[0].count + 1
      Tag.find(tag[0]).update_attributes(name:name, count: count)
      tag_id = tag[0].id
    end
    PostTag.create(post_id: post_id, tag_id: tag_id) 
  end
end
